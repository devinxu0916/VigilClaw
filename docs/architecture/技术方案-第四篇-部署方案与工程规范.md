**关键配置说明**：

| 配置                       | 值                                       | 理由                        |
| -------------------------- | ---------------------------------------- | --------------------------- |
| `target: ES2022`           | 使用 top-level await, structuredClone 等 | Node 22 完整支持            |
| `module: Node16`           | ESM + CJS 互操作                         | 与 Node.js 原生模块系统对齐 |
| `strict: true`             | 全部严格检查                             | 类型安全是设计原则          |
| `noUncheckedIndexedAccess` | 数组/对象索引返回 `T \| undefined`       | 防止运行时 undefined 错误   |

| 依赖                        | 用途                | 大小                 | 许可证     | 替代方案                        |
| --------------------------- | ------------------- | -------------------- | ---------- | ------------------------------- |
| `@anthropic-ai/sdk`         | Claude API 客户端   | ~200KB               | MIT        | 手写 HTTP（不值得）             |
| `better-sqlite3`            | SQLite 驱动         | ~2MB (native)        | MIT        | sqlite3（性能差）               |
| `sqlite-vec`                | SQLite 向量搜索扩展 | ~1MB (native)        | MIT        | 无替代（唯一 SQLite 向量方案）  |
| `@huggingface/transformers` | 本地嵌入模型        | ~500KB (+80MB model) | Apache-2.0 | API 嵌入（增加外部依赖）        |
| `grammy`                    | Telegram Bot        | ~300KB               | MIT        | node-telegram-bot-api（类型差） |
| `dockerode`                 | Docker API          | ~150KB               | Apache-2.0 | 手写 Docker Socket（不值得）    |
| `pino`                      | 结构化日志          | ~200KB               | MIT        | winston（更大）                 |
| `zod`                       | Schema 校验         | ~60KB                | MIT        | joi（更大）/ 手写（不类型安全） |
| `cron-parser`               | Cron 表达式         | ~30KB                | MIT        | 手写解析（易出错）              |
| **合计**                    | **9 个**            | **~4.5MB**           |            |                                 |

> **注**：`uuid` 已移除，使用 `crypto.randomUUID()` 替代。Phase 2 新增 `sqlite-vec` 和 `@huggingface/transformers` 用于持久记忆功能（从 `@xenova/transformers` 迁移，解决 sharp 依赖问题）。

### 4.2.4 依赖治理规则

| 规则                   | 执行方式                              |
| ---------------------- | ------------------------------------- |
| 新增依赖必须有充分理由 | PR Description 中说明为什么不能手写   |

```typescript
    expect(parsed.messages).toHaveLength(1);
  });

  it('should detect output file creation via fs.watch', async () => {
    const resultPath = path.join(ipcDir, 'output', 'result-test-123.json');

      const watcher = fs.watch(path.join(ipcDir, 'output'), (event, filename) => {

          resolve(filename);
```

### 4.3.4 E2E 安全测试

```typescript
// tests/e2e/network-security.test.ts
// 需要 Docker 环境，CI 中运行
import { describe, it, expect, beforeAll, afterAll } from 'vitest';
import Docker from 'dockerode';

describe('Container Network Security (E2E)', { timeout: 60_000 }, () => {
  const docker = new Docker();
  let container: Docker.Container;

  beforeAll(async () => {
    // 构建测试镜像
    container = await docker.createContainer({
      Image: 'vigilclaw/agent-runner:latest',
      Cmd: ['sleep', '30'],
      HostConfig: {
        NetworkMode: 'none',
        ReadonlyRootfs: true,
        Tmpfs: { '/tmp': 'rw,noexec,nosuid,size=10m' },
        AutoRemove: true,
      },
    });
    await container.start();
  });

  afterAll(async () => {
    try {
      await container.stop({ t: 1 });
    } catch {
      /* already stopped */
    }
  });

  async function exec(cmd: string[]): Promise<{ stdout: string; exitCode: number }> {
    const execObj = await container.exec({
      Cmd: cmd,
      AttachStdout: true,
      AttachStderr: true,
    });
    const stream = await execObj.start({});
    const chunks: Buffer[] = [];
    for await (const chunk of stream) chunks.push(chunk as Buffer);
    const result = await execObj.inspect();
    return { stdout: Buffer.concat(chunks).toString(), exitCode: result.ExitCode ?? 1 };
  }

  it('should not have any network interfaces (NetworkMode: none)', async () => {
    const { stdout } = await exec(['ip', 'addr']);
    // 只有 lo (loopback)
    expect(stdout).toContain('lo');
    expect(stdout).not.toContain('eth0');
  });

  it('should not expose API key via environment', async () => {
    const { stdout } = await exec(['printenv']);
    expect(stdout).not.toContain('ANTHROPIC_API_KEY');
    expect(stdout).not.toContain('sk-ant-');
  });

  it('should have read-only root filesystem', async () => {
    const { exitCode } = await exec(['sh', '-c', 'touch /test-readonly 2>&1']);
    expect(exitCode).not.toBe(0);
  });

  it('should allow writing to /tmp', async () => {
    const { exitCode } = await exec(['sh', '-c', 'echo test > /tmp/test-write']);
    expect(exitCode).toBe(0);
  });

  it('should not allow executing from /tmp (noexec)', async () => {
    await exec([
      'sh',
      '-c',
      'echo "#!/bin/sh\necho pwned" > /tmp/evil.sh && chmod +x /tmp/evil.sh',
    ]);
    const { exitCode } = await exec(['sh', '-c', '/tmp/evil.sh 2>&1']);
    expect(exitCode).not.toBe(0);
  });
});
```

```yaml
    ports: [] # 无需暴露端口（Telegram 用 polling）

      - no-new-privileges:true
```

### 4.6.2 健康检查

```typescript
// src/health.ts
import http from 'node:http';

export function startHealthServer(port: number, db: Database.Database, docker: Docker): void {
  const server = http.createServer(async (req, res) => {
    if (req.url !== '/health') {
      res.writeHead(404);
      res.end();
      return;
    }

    const checks = {
      sqlite: checkSqlite(db),
      docker: await checkDocker(docker),
      uptime: process.uptime(),
      memory: process.memoryUsage().rss,
    };

    const healthy = checks.sqlite && checks.docker;
    res.writeHead(healthy ? 200 : 503, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify({ status: healthy ? 'ok' : 'degraded', checks }));
  });

  // 仅监听 localhost（安全）
  server.listen(port, '127.0.0.1', () => {
    logger.info({ port }, 'Health check server started');
  });
}

function checkSqlite(db: Database.Database): boolean {
  try {
    db.prepare('SELECT 1').get();
    return true;
  } catch {
    return false;
  }
}

async function checkDocker(docker: Docker): Promise<boolean> {
  try {
    await docker.ping();
    return true;
  } catch {
    return false;
  }
}
```

```json
  }
}
```
